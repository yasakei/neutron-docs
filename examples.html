Classes and Objects

Object-oriented programming with classes

person.nt 
class Person {
    var name;

    fun setName(name) {
        this.name = name;
    }

    fun greet() {
        print("Hello, my name is " + this.name);
    }
}

var person = Person();
person.setName("Alice");
person.greet();

// Output: Hello, my name is Alice

Simple Calculator

Basic arithmetic operations

calculator.nt 
class Calculator {
    fun add(a, b) {
        return a + b;
    }
    
    fun subtract(a, b) {
        return a - b;
    }
    
    fun multiply(a, b) {
        return a * b;
    }
    
    fun divide(a, b) {
        if (b != 0) {
            return a / b;
        } else {
            return "Error: Division by zero";
        }
    }
}

var calc = Calculator();
print(calc.add(10, 5));      // 15
print(calc.subtract(10, 3)); // 7
print(calc.multiply(4, 6));  // 24
print(calc.divide(20, 4));   // 5

Working with Modules

Using the standard library modules

modules.nt 
// Import math module
import "math";

var radius = 5;
var area = math.PI * math.pow(radius, 2);
print("Circle area: " + area);

var squareRoot = math.sqrt(16);
print("Square root of 16: " + squareRoot);

// Import JSON module
import "json";

var data = json.parse('{"name": "Neutron", "version": 1.0}');
print("Language: " + data.name);
print("Version: " + data.version);

Fibonacci Sequence

Recursive function example

fibonacci.nt 
fun fibonacci(n) {
    if (n <= 1) {
        return n;
    } else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

// Print first 10 Fibonacci numbers
for (var i = 0; i < 10; i = i + 1) {
    print("F(" + i + ") = " + fibonacci(i));
}

// Output:
// F(0) = 0
// F(1) = 1
// F(2) = 1
// F(3) = 2
// F(4) = 3
// F(5) = 5
// F(6) = 8
// F(7) = 13
// F(8) = 21
// F(9) = 34
